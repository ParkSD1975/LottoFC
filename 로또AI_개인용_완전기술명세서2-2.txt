/**
 * í†µê³„ ê³„ì‚° í•¨ìˆ˜
 * v1.1.1ì—ì„œ ì‹ ê·œ ì¶”ê°€
 */
export function calculateStatistics(values: number[]) {
  const sorted = [...values].sort((a, b) => a - b)
  
  // ìµœì†Œ, ìµœëŒ€
  const min = sorted[0]
  const max = sorted[sorted.length - 1]
  
  // í‰ê· 
  const mean = values.reduce((a, b) => a + b, 0) / values.length
  
  // ì¤‘ì•™ê°’
  const mid = Math.floor(sorted.length / 2)
  const median = sorted.length % 2 === 0
    ? (sorted[mid - 1] + sorted[mid]) / 2
    : sorted[mid]
  
  // ìµœë¹ˆê°’
  const frequency: Record<number, number> = {}
  values.forEach((v) => {
    frequency[v] = (frequency[v] || 0) + 1
  })
  const mode = parseInt(
    Object.entries(frequency).sort((a, b) => b[1] - a[1])[0][0]
  )
  
  // í‘œì¤€í¸ì°¨
  const variance = values.reduce((acc, v) => acc + Math.pow(v - mean, 2), 0) / values.length
  const stdDev = Math.sqrt(variance)
  
  return {
    min,
    max,
    mean,
    median,
    mode,
    stdDev,
  }
}

/**
 * ë¹„ìœ¨ í†µê³„ ê³„ì‚° (í™€ì§, ì €ê³  ë“±)
 */
export function calculateRatioStatistics(
  data: Array<{ ratio: string; count: number }>
) {
  const total = data.reduce((acc, d) => acc + d.count, 0)
  
  const ratios: Record<string, number> = {}
  data.forEach((d) => {
    ratios[d.ratio] = d.count
  })
  
  const mostCommon = data.sort((a, b) => b.count - a.count)[0]
  
  return {
    ratios,
    mostCommon: mostCommon.ratio,
    mostCommonCount: mostCommon.count,
    mostCommonPercent: (mostCommon.count / total) * 100,
  }
}
